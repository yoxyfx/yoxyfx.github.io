window.scrollTo() 및 smooth 동작: 스크롤 위치를 "top: 0"으로 지정하여 .fixed-button 버튼을 클릭하면 페이지가 부드럽게 최상단으로 스크롤되도록 기능을 구현 

                
document.querySelector('.fixed-button').addEventListener('click', function() {
    window.scrollTo({ top: 0, behavior: 'smooth' });
});

setTimeout(), IntersectionObserver: 게setTimeout()을 사용하여 각 이미지를 순차적으로 등장시키는 효과를 구현'
IntersectionObserver를 사용해 이미지들이 화면에 보일 때 애니메이션을 시작
.image-container2 img 섹션의 이미지를 300ms 간격으로 차례대로 나타나도록 설정 

                
document.addEventListener("DOMContentLoaded", function () {
    const section = document.getElementById("problemsolution");
    const images = document.querySelectorAll(".image-container2 img");
    function revealImages() {
        const sectionTop = section.getBoundingClientRect().top;
        const windowHeight = window.innerHeight;
        if (sectionTop < windowHeight) {
            images.forEach((img, index) => {
                setTimeout(() => {
                    img.classList.add("show");
                }, index * 300);
            });
            window.removeEventListener("scroll", revealImages);
        }
    }
    window.addEventListener("scroll", revealImages);
});

IntersectionObserver API: 스크롤 시 animate 클래스가 붙은 요소 애니메이션
.animate 클래스를 가진 요소가 화면에 보일 때 visible 클래스를 추가하여 애니메이션을 적용 

                
const visibilityObserver = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
        if (entry.isIntersecting) {
            entry.target.classList.add('visible');
        }
    });
});
const elementsToAnimate = document.querySelectorAll('.animate');
elementsToAnimate.forEach(element => {
    visibilityObserver.observe(element);
});
