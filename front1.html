Circle 클래스 정의: 원의 이동속도 및 방향 조절, 원 위치 업데이트

                
function Circle(x, y) {
    this.x = x;
    this.y = y;
    this.radius = random(10, 30);
    this.color = `hsl(${Math.random() * 360}, 100%, 50%)`;
    this.dx = random(-2, 2);
    this.dy = random(-2, 2);
    
    this.draw = () => {
        ctx.beginPath();
        ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2, false);
        ctx.fillStyle = this.color;
        ctx.fill();
        ctx.closePath();
    };
    
    this.update = () => {
        this.x += this.dx;
        this.y += this.dy;
    
        if (this.x + this.radius > width || this.x - this.radius < 0) this.dx = -this.dx;
        if (this.y + this.radius > height || this.y - this.radius < 0) this.dy = -this.dy;
    
        this.draw();
    };
    }

퍼즐

초기 배치: setInitialPositions() 함수가 각 퍼즐 조각을 container 내 랜덤 위치에 배치
드래그 앤 드롭: dragStart: 드래그된 조각의 ID 저장.
drop: 마우스 위치에 조각을 이동.
dragover: 드롭 이벤트를 허용.
재시작: Restart 버튼 클릭 시, 조각이 다시 초기 위치로 배치됩니다. 

                
const pieces = document.querySelectorAll('.piece:not(#img6)');
const restartButton = document.getElementById('restart');

function setInitialPositions() {
    pieces.forEach(piece => {
        piece.style.left = `${Math.random() * 500}px`;
        piece.style.top = `${Math.random() * 500}px`;
    });
}

function dragStart(event) {
    event.dataTransfer.setData('text/plain', event.target.id);
}

function drop(event) {
    event.preventDefault();
    const id = event.dataTransfer.getData('text/plain');
    const piece = document.getElementById(id);
    piece.style.left = `${event.offsetX - 125}px`;
    piece.style.top = `${event.offsetY - 125}px`;
}

document.getElementById('container').addEventListener('dragover', e => e.preventDefault());
document.getElementById('container').addEventListener('drop', drop);
restartButton.addEventListener('click', setInitialPositions);
