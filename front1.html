픽셀 그리기 (drawPixel함수): 주어진 좌표(x, y)에 픽셀을 색칠합니다.
색상은 color 매개변수를 통해 지정됩니다.(사용자의 움직임을 시각적으로 표현하는 데 사용) 

                
function drawPixel(x, y, color) {
    ctx.fillStyle = color;
    ctx.fillRect(x, y, pixelSize, pixelSize);
}

타이머 및 라운드 관리 (countdown 함수): 게임의 타이머를 관리하며, 시간이 0에 도달하거나 모든 픽셀이 파란색으로 채워지면 라운드를 변경하고 게임을 종료 라운드가 진행될수록 캔버스의 크기가 증가하고 픽셀의 크기가 감소하여 난이도가 조절

                
function countdown() {
    time--;
    timeDisplay.textContent = time;

    if (time <= 0) {
        clearInterval(timerId);
        playSound(roundEndSound);
        if (round < totalRounds) {
            alert('Time\'s up! Moving to the next round.');
            round++;
            roundDisplay.textContent = round;
            roundScores[round - 2] = score; 
            reset();
        } else {
            endGame();
        }
    }

게임 초기화 (reset 함수): 주어진 좌표(x, y)의 픽셀을 지웁니다.
게임을 초기 상태로 되돌리는 기능으로, 시간, 점수, 캔버스 크기 등을 초기화하고 새 게임을 시작할 준비 

                
function erasePixel(x, y) {
    ctx.clearRect(x, y, pixelSize, pixelSize);
}

게임 시작 버튼 (startButton이벤트 리스너): 사용자가 'startButton'을 클릭하면 게임이 시작됨- 사용자로부터 이름을 입력받고, 환영 메시지를 표시-타이머를 시작하고 게임의 상태를 '시작됨＇으로 설정 

                
function erasePixel(x, y) {
    ctx.clearRect(x, y, pixelSize, pixelSize);
}

키보드 입력 처리 (키보드 이벤트 리스너): 주사용자의 입력(화살표 키)에 따라 사용자의 위치를 업데이트하고 캔버스에 그림을 그리는 `drawLine` 함수를 호출-스페이스바를 누르면 게임이 초기화됩니다. 

                
document.addEventListener('keydown', function(event) {
    if (event.key === ' ') {
        reset();
        playSound(resetSound);
    } else {
        drawLine(event.key);
    }
});

총 픽셀 수 계산 (getTotalPixels함수): 캔버스의 가능한 픽셀 수를 계산-사용자가 모든 픽셀을 채웠는지 확인하는 데 사용됩니다. 

                    
const totalPixels = getTotalPixels();
if (bluePixelCount === totalPixels) {
    clearInterval(timerId);
    playSound(roundEndSound);
    if (round < totalRounds) {
        alert(round + ' 라운드 종료! 당신의 점수: ' + score);
        roundScores[round - 1] = score; 
        round++;
        roundDisplay.textContent = round;
        reset();
    } else {
        endGame();
    }
}

                    
                
                
function getTotalPixels() {
    return (canvas.width / size) * (canvas.height / size);
}
